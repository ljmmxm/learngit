安装windows git后。

进入gitbash

#配置git
//配置名字
$git config --global user.name "your name"
//配置邮件
$git config --global user.email "xxx@xxx.com"

#创建git版本库
//创建版本库目录
$mkdir learngit
//进入版本库目录
$cd learngit
//查看当前目录绝对路径
$pwd

#初始换版本库
$git init

#添加到仓库
$git add xxx

#提交到仓库
$git commit -m "描述信息"

#查看仓库状态
$git status

#查看修改内容
$git diff xxx

#查看日志
$git log

#查看日志简略信息
$git log --pretty=oneline

#回滚到指定版本
$git reset --hard xxx

#查询命令历史
$git reflog
